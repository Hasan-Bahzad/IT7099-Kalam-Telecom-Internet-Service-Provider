 network 74.211.4.20 0.0.0.3

 network 120.120.120.120 0.0.0.0
 network 172.24.10.0 0.0.0.255

 network 172.24.30.0 0.0.0.255

 key 1
  key-string XYZ-Kalam20

interface Serial1/0

 ip authentication key-chain eigrp 20 XYZ_AuthenKey
 ip hello-interval eigrp 100 10
 ip hold-time eigrp 100 30

interface Serial1/1
